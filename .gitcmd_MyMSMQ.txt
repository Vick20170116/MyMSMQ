d:
cd D:\git\MyMSMQ
echo "# MyMSMQ" >> README.md


#建立local resposity (建立.git目錄,commit錯可還原此目錄)
git init  
#新增或修改時須執行(git add .)把workspace資料加到staging
git add .
#顯示狀態(顯示異動哪些檔案)
git status
#把staging 資料commit到 local Repo ,並加上備註
git commit -m "first commit"
#rem MyMSMQ since 20160827
#將本地的respo與遠端MyMSMQ respo 結合
git remote add MyMSMQ https://github.com/michaelfangtw/MyMSMQ.git
#顯示
git remote 
#遠端比較新時,須更新本地端
#遠端若刪除本地也會刪除
#git pull=git fetch+git merge   (檢視遠端異動,並更新到本地)
#git pull

#更新到遠端Resposity
git push --set-upstream MyMSMQ master
